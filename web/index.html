function show(id){ ['ask-card','chat-card','flash-card'].forEach(c=> document.getElementById(c).style.display = 'none'); document.getElementById(id).style.display = 'block'; } document.getElementById('tab-ask').addEventListener('click', ()=> show('ask-card')); document.getElementById('tab-chat').addEventListener('click', ()=> show('chat-card')); document.getElementById('tab-flash').addEventListener('click', ()=> show('flash-card')); // Chat async function chat(){ const m = document.getElementById('chat-input').value; const res = await fetch('/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message: m }) }); const data = await res.json(); const ans = document.getElementById('chat-ans'); ans.textContent = data.answer || 'No answer'; ans.hidden = false; } document.getElementById('chat-btn').addEventListener('click', chat); document.getElementById('chat-input').addEventListener('keydown', e => { if (e.key === 'Enter') chat(); }); // Flashcards async function loadFlash(){ const res = await fetch('/flashcards'); const data = await res.json(); const list = document.getElementById('flash-list'); list.innerHTML = ''; data.forEach(f => { const el = document.createElement('div'); el.style.padding='8px'; el.style.borderBottom='1px solid #eee'; el.innerHTML = `Q: ${f.question}
A: ${f.answer}`; list.appendChild(el); }); } document.getElementById('load-flash').addEventListener('click', loadFlash);
